    function toggle_textarea(element) {
      var div = document.getElementById(element);

        div.style.display = 'block';
        div.style.visibility = 'visible';
        var textBox = document.getElementById(element + '_text');
        textBox.select();
      } else {
        div.style.display = 'none';
      }
    }
    function toggle_help(element, tip_element) {
      var div = document.getElementById(element);
      var tip_div = document.getElementById(tip_element);
      tip_div.style.display = 'none';
      if (div.style.display == 'none') {
        div.style.display = 'block';
        div.style.visibility = 'visible';
      } else {
        div.style.display = 'none';
      }
    }
    function hover_show(element) {
      var div = document.getElementById(element);
      div.style.display = 'inline-block';
      div.style.visibility = 'visible';
    }
    function hover_vanish(element) {
      var div = document.getElementById(element);
